The request was sent to a server that isn't configured to produce a response for that specific destination.

## When and Why to Use It
Used mostly by complex CDN/Server infrastructures when a request is accidentally routed to the wrong internal server.

### Usage Examples

<Tabs>
  <TabItem label="Shokupan (Functional)">

```typescript
import { HttpException, get } from 'shokupan';

export const getResource = get('/resource', () => {
    throw new HttpException('Misdirected Request', 421);
});
```

  </TabItem>
  <TabItem label="Shokupan (Controller)">

```typescript
import { Controller, Get, HttpException } from 'shokupan';

@Controller('/api')
export class ExampleController {
    @Get('/resource')
    getResource() {
        throw new HttpException('Misdirected Request', 421);
    }
}
```

  </TabItem>
  <TabItem label="Express">

```typescript
import express from 'express';
const app = express();

app.get('/resource', (req, res) => {
    res.status(421).json({ error: 'Misdirected Request' });
});
```

  </TabItem>
  <TabItem label="Nest">

```typescript
import { Controller, Get, HttpException } from '@nestjs/common';

@Controller('api')
export class ExampleController {
  @Get('resource')
  getResource() {
    throw new HttpException('Misdirected Request', 421);
  }
}
```

  </TabItem>
  <TabItem label="Koa">

```typescript
import Koa from 'koa';
const app = new Koa();

app.use(async ctx => {
  if (ctx.path === '/resource' && ctx.method === 'GET') {
    ctx.status = 421;
    ctx.body = { error: 'Misdirected Request' };
  }
});
```
