### Usage Examples

<Tabs>
  <TabItem label="Shokupan (Functional)">

```typescript
import { HttpException, get } from 'shokupan';

export const getResource = get('/resource', () => {
    throw new HttpException('Origin Is Unreachable', 523);
});
```

  </TabItem>
  <TabItem label="Shokupan (Controller)">

```typescript
import { Controller, Get, HttpException } from 'shokupan';

@Controller('/api')
export class ExampleController {
    @Get('/resource')
    getResource() {
        throw new HttpException('Origin Is Unreachable', 523);
    }
}
```

  </TabItem>
  <TabItem label="Express">

```typescript
import express from 'express';
const app = express();

app.get('/resource', (req, res) => {
    res.status(523).json({ error: 'Origin Is Unreachable' });
});
```

```typescript
  }

```

  <TabItem label="Koa">

```typescript
import Koa from 'koa';
const app = new Koa();

app.use(async ctx => {
  if (ctx.path === '/resource' && ctx.method === 'GET') {
    ctx.status = 523;
    ctx.body = { error: 'Origin Is Unreachable' };
  }
});
```
